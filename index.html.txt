        function calculateMortgage() {
            const price = parseFloat(homePrice.value);
            const downPercent = parseFloat(downPayment.value);
            const annualRate = parseFloat(interestRate.value);
            const loanTermYears = parseInt(document.getElementById('loanTerm').value);
            const propertyTax = parseFloat(document.getElementById('propertyTax').value) || 0;
            const insurance = parseFloat(document.getElementById('insurance').value) || 0;
            const loanAmount = price - (price * (downPercent / 100));
            const monthlyRate = annualRate / 100 / 12;
            const n = loanTermYears * 12;
            const monthlyPayment = monthlyRate === 0 ? loanAmount / n : loanAmount * (monthlyRate * Math.pow(1 + monthlyRate, n)) / (Math.pow(1 + monthlyRate, n) - 1);
            const totalPayment = monthlyPayment + (propertyTax / 12) + (insurance / 12);
            const totalInterest = (monthlyPayment * n) - loanAmount;
            document.getElementById('resultLoanAmount').textContent = '$' + Math.round(loanAmount).toLocaleString();
            document.getElementById('resultMonthlyPayment').textContent = '$' + Math.round(monthlyPayment).toLocaleString();
            document.getElementById('resultMonthlyTax').textContent = '$' + Math.round(propertyTax / 12).toLocaleString();
            document.getElementById('resultMonthlyInsurance').textContent = '$' + Math.round(insurance / 12).toLocaleString();
            document.getElementById('resultTotalPayment').textContent = '$' + Math.round(totalPayment).toLocaleString();
            document.getElementById('resultTotalInterest').textContent = '$' + Math.round(totalInterest).toLocaleString();
            document.getElementById('calculatorResults').classList.add('active');
        }
